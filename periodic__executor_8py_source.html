114 # _EXECUTORS has a weakref to each running PeriodicExecutor. Once started,
115 # an executor is kept alive by a strong reference from its thread and perhaps
116 # from other objects. When the thread dies and all other referrers are freed,
117 # the executor is freed and removed from _EXECUTORS. If any threads are
118 # running when the interpreter begins to shut down, we try to halt and join
119 # them to avoid spurious errors.
120 _EXECUTORS = set()
