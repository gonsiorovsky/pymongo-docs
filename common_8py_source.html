75 def partition_node(node):
76  """Split a host:port string into (host, int(port)) pair."""
77  host = node
78  port = 27017
79  idx = node.rfind(':')
80  if idx != -1:
81  host, port = node[:idx], int(node[idx + 1:])
82  if host.startswith('['):
83  host = host[1:-1]
84  return host, port

87 def clean_node(node):
88  """Split and normalize a node name from an ismaster response."""
