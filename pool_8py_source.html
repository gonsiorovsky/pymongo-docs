298  def _raise_connection_failure(self, error):
299  # Catch *all* exceptions from socket methods and close the socket. In
300  # regular Python, socket operations only raise socket.error, even if
301  # the underlying cause was a Ctrl-C: a signal raised during socket.recv
302  # is expressed as an EINTR error from poll. See internal_select_ex() in
303  # socketmodule.c. All error codes from poll become socket.error at
304  # first. Eventually in PyEval_EvalFrameEx the interpreter checks for
305  # signals and throws KeyboardInterrupt into the current frame on the
306  # main thread.
307  #
308  # But in Gevent and Eventlet, the polling mechanism (epoll, kqueue,
309  # ...) is called in Python code, which experiences the signal as a
310  # KeyboardInterrupt from the start, rather than as an initial
311  # socket.error, so we catch that, close the socket, and reraise it.
312  self.close()
313  if isinstance(error, socket.error):
314  _raise_connection_failure(self.address, error)
315  else:
316  raise error

335 def _create_connection(address, options):
336  """Given (host, port) and PoolOptions, connect and return a socket object.
337 
338  Can raise socket.error.
339 
340  This is a modified version of create_connection from CPython >= 2.6.
341  """
342  host, port = address
343 
344  # Check if dealing with a unix domain socket
345  if host.endswith('.sock'):
346  if not hasattr(socket, "AF_UNIX"):
347  raise ConnectionFailure("UNIX-sockets are not supported "
348  "on this system")
349  sock = socket.socket(socket.AF_UNIX)
350  try:
351  sock.connect(host)
352  return sock
353  except socket.error:
354  sock.close()
355  raise
356 
357  # Don't try IPv6 if we don't support it. Also skip it if host
358  # is 'localhost' (::1 is fine). Avoids slow connect issues
359  # like PYTHON-356.
360  family = socket.AF_INET
361  if socket.has_ipv6 and host != 'localhost':
362  family = socket.AF_UNSPEC
363 
364  err = None
365  for res in socket.getaddrinfo(host, port, family, socket.SOCK_STREAM):
366  af, socktype, proto, dummy, sa = res
367  sock = socket.socket(af, socktype, proto)
368  try:
369  sock.setsockopt(socket.IPPROTO_TCP, socket.TCP_NODELAY, 1)
370  sock.settimeout(options.connect_timeout)
371  sock.setsockopt(socket.SOL_SOCKET, socket.SO_KEEPALIVE,
372  options.socket_keepalive)
373  sock.connect(sa)
374  return sock
375  except socket.error as e:
376  err = e
377  sock.close()
378 
379  if err is not None:
380  raise err
381  else:
382  # This likely means we tried to connect to an IPv6 only
383  # host with an OS/kernel or Python interpreter that doesn't
384  # support IPv6. The test case is Jython2.5.1 which doesn't
385  # support IPv6 at all.
386  raise socket.error('getaddrinfo failed')

389 def _configured_socket(address, options):
390  """Given (host, port) and PoolOptions, return a configured socket.
391 
392  Can raise socket.error, ConnectionFailure, or CertificateError.
393 
394  Sets socket's SSL and timeout options.
395  """
396  sock = _create_connection(address, options)
397  ssl_context = options.ssl_context
398 
399  if ssl_context is not None:
400  try:
401  sock = ssl_context.wrap_socket(sock)
402  except IOError as exc:
403  sock.close()
404  raise ConnectionFailure("SSL handshake failed: %s" % (str(exc),))
405  if ssl_context.verify_mode and options.ssl_match_hostname:
406  try:
407  match_hostname(sock.getpeercert(), hostname=address[0])
408  except CertificateError:
409  sock.close()
410  raise
411 
412  sock.settimeout(options.socket_timeout)
413  return sock

419  def __init__(self, address, options, handshake=True):
420  """
421  :Parameters:
422  - `address`: a (hostname, port) tuple
423  - `options`: a PoolOptions instance
424  - `handshake`: whether to call ismaster for each new SocketInfo
425  """
426  # Check a socket's health with socket_closed() every once in a while.
427  # Can override for testing: 0 to always check, None to never check.
428  self._check_interval_seconds = 1
429 
430  self.sockets = set()
431  self.lock = threading.Lock()
432 
433  # Keep track of resets, so we notice sockets created before the most
434  # recent reset and close them.
435  self.pool_id = 0
436  self.pid = os.getpid()
437  self.address = address
438  self.opts = options
439  self.handshake = handshake
440 
441  if (self.opts.wait_queue_multiple is None or
442  self.opts.max_pool_size is None):
443  max_waiters = None
444  else:
445  max_waiters = (
446  self.opts.max_pool_size * self.opts.wait_queue_multiple)
447 
448  self._socket_semaphore = thread_util.create_semaphore(
449  self.opts.max_pool_size, max_waiters)
450 
451  def reset(self):
452  with self.lock:
453  self.pool_id += 1
454  self.pid = os.getpid()
455  sockets, self.sockets = self.sockets, set()
456 
457  for sock_info in sockets:
458  sock_info.close()

460  def connect(self):
461  """Connect to Mongo and return a new SocketInfo.
462 
463  Can raise ConnectionFailure or CertificateError.
464 
465  Note that the pool does not keep a reference to the socket -- you
466  must call return_socket() when you're done with it.
467  """

469  try:
470  sock = _configured_socket(self.address, self.opts)
471  if self.handshake:
472  ismaster = IsMaster(command(sock, 'admin', {'ismaster': 1},

474  ReadPreference.PRIMARY,
475  DEFAULT_CODEC_OPTIONS))

477  ismaster = None
478  return SocketInfo(sock, self, ismaster, self.address)
479  except socket.error as error:
480  if sock is not None:
481  sock.close()

484  @contextlib.contextmanager
485  def get_socket(self, all_credentials, checkout=False):
486  """Get a socket from the pool. Use with a "with" statement.
487 
488  Returns a :class:`SocketInfo` object wrapping a connected
489  :class:`socket.socket`.
490 
491  This method should always be used in a with-statement::
492 
493  with pool.get_socket(credentials, checkout) as socket_info:
494  socket_info.send_message(msg)
495  data = socket_info.receive_message(op_code, request_id)
496 
497  The socket is logged in or out as needed to match ``all_credentials``
498  using the correct authentication mechanism for the server's wire
499  protocol version.
500 
501  Can raise ConnectionFailure or OperationFailure.
502 
503  :Parameters:
504  - `all_credentials`: dict, maps auth source to MongoCredential.
505  - `checkout` (optional): keep socket checked out.
506  """
507  # First get a socket, then attempt authentication. Simplifies
508  # semaphore management in the face of network errors during auth.
509  sock_info = self._get_socket_no_auth()
510  try:
511  sock_info.check_auth(all_credentials)
512  yield sock_info
513  except:
514  # Exception in caller. Decrement semaphore.
515  self.return_socket(sock_info)
516  raise
517  else:
518  if not checkout:
519  self.return_socket(sock_info)

521  def _get_socket_no_auth(self):
522  """Get or create a SocketInfo. Can raise ConnectionFailure."""
523  # We use the pid here to avoid issues with fork / multiprocessing.
524  # See test.test_client:TestClient.test_fork for an example of
525  # what could go wrong otherwise
526  if self.pid != os.getpid():
527  self.reset()
528 
529  # Get a free socket or create one.
530  if not self._socket_semaphore.acquire(
531  True, self.opts.wait_queue_timeout):
532  self._raise_wait_queue_timeout()
533 
534  # We've now acquired the semaphore and must release it on error.
535  try:
536  try:
537  # set.pop() isn't atomic in Jython less than 2.7, see
538  # http://bugs.jython.org/issue1854
539  with self.lock:
540  sock_info, from_pool = self.sockets.pop(), True
541  except KeyError:
542  # Can raise ConnectionFailure or CertificateError.
543  sock_info, from_pool = self.connect(), False
544 
545  if from_pool:
546  # Can raise ConnectionFailure.
547  sock_info = self._check(sock_info)
548 
549  except:
550  self._socket_semaphore.release()
551  raise
552 
553  sock_info.last_checkout = _time()
554  return sock_info

556  def return_socket(self, sock_info):
557  """Return the socket to the pool, or if it's closed discard it."""
558  if self.pid != os.getpid():
559  self.reset()
560  else:
561  if sock_info.pool_id != self.pool_id:
562  sock_info.close()
563  elif not sock_info.closed:
564  with self.lock:
565  self.sockets.add(sock_info)
566 
567  self._socket_semaphore.release()

569  def _check(self, sock_info):
570  """This side-effecty function checks if this pool has been reset since
571  the last time this socket was used, or if the socket has been closed by
572  some external network error, and if so, attempts to create a new socket.
573  If this connection attempt fails we reset the pool and reraise the
574  ConnectionFailure.
575 
576  Checking sockets lets us avoid seeing *some*
577  :class:`~pymongo.errors.AutoReconnect` exceptions on server
578  hiccups, etc. We only do this if it's been > 1 second since
579  the last socket checkout, to keep performance reasonable - we
580  can't avoid AutoReconnects completely anyway.
581  """
582  error = False
583 
584  # How long since socket was last checked out.
585  age = _time() - sock_info.last_checkout
586  if (self._check_interval_seconds is not None
587  and (
588  0 == self._check_interval_seconds
589  or age > self._check_interval_seconds)):
590  if socket_closed(sock_info.sock):
591  sock_info.close()
592  error = True
593 
594  if not error:
595  return sock_info
596  else:
597  return self.connect()
