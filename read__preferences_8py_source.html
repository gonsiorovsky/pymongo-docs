41 def _validate_tag_sets(tag_sets):
42  """Validate tag sets for a MongoReplicaSetClient.
43  """
44  if tag_sets is None:
45  return tag_sets
46 
47  if not isinstance(tag_sets, list):
48  raise TypeError((
49  "Tag sets %r invalid, must be a list") % (tag_sets,))
50  if len(tag_sets) == 0:
51  raise ValueError((
52  "Tag sets %r invalid, must be None or contain at least one set of"
53  " tags") % (tag_sets,))
54 
55  for tags in tag_sets:
56  if not isinstance(tags, Mapping):
57  raise TypeError(
58  "Tag set %r invalid, must be an instance of dict, "
59  "bson.son.SON or other type that inherits from "
60  "collection.Mapping" % (tags,))
61 
62  return tag_sets

65 class _ServerMode(object):
66  """Base class for all read preferences.
67  """
68 
69  __slots__ = ("__mongos_mode", "__mode", "__tag_sets")
70 
71  def __init__(self, mode, tag_sets=None):
72  if mode == _PRIMARY and tag_sets is not None:
73  raise ConfigurationError("Read preference primary "
74  "cannot be combined with tags")
75  self.__mongos_mode = _MONGOS_MODES[mode]
76  self.__mode = mode
77  self.__tag_sets = _validate_tag_sets(tag_sets)
78 
79  @property
80  def name(self):
81  """The name of this read preference.
82  """
83  return self.__class__.__name__
84 
85  @property
86  def document(self):
87  """Read preference as a document.
88  """
89  if self.__tag_sets in (None, [{}]):
90  return {'mode': self.__mongos_mode}
91  return {'mode': self.__mongos_mode, 'tags': self.__tag_sets}
92 
93  @property
94  def mode(self):
95  """The mode of this read preference instance.
96  """
97  return self.__mode
98 
99  @property
100  def tag_sets(self):
101  """Set ``tag_sets`` to a list of dictionaries like [{'dc': 'ny'}] to
102  read only from members whose ``dc`` tag has the value ``"ny"``.
103  To specify a priority-order for tag sets, provide a list of
104  tag sets: ``[{'dc': 'ny'}, {'dc': 'la'}, {}]``. A final, empty tag
105  set, ``{}``, means "read from any member that matches the mode,
106  ignoring tags." MongoReplicaSetClient tries each set of tags in turn
107  until it finds a set of tags with at least one matching member.
108 
109  .. seealso:: `Data-Center Awareness
110  <http://www.mongodb.org/display/DOCS/Data+Center+Awareness>`_
111  """
112  return list(self.__tag_sets) if self.__tag_sets else [{}]
113 
114  def __repr__(self):
115  return "%s(tag_sets=%r)" % (
116  self.name, self.__tag_sets)
117 
118  def __eq__(self, other):
119  if isinstance(other, _ServerMode):
120  return (self.mode == other.mode and
121  self.tag_sets == other.tag_sets)
122  raise NotImplementedError
123 
124  def __ne__(self, other):
125  return not self == other

281 class ReadPreference(object):
282  """An enum that defines the read preference modes supported by PyMongo.
283 
284  See :doc:`/examples/high_availability` for code examples.
285 
286  A read preference is used in three cases:
287 
288  :class:`~pymongo.mongo_client.MongoClient` connected to a single mongod:
289 
290  - ``PRIMARY``: Queries are allowed if the server is standalone or a replica
291  set primary.
292  - All other modes allow queries to standalone servers, to a replica set
293  primary, or to replica set secondaries.
294 
295  :class:`~pymongo.mongo_client.MongoClient` initialized with the
296  ``replicaSet`` option:
297 
298  - ``PRIMARY``: Read from the primary. This is the default, and provides the
299  strongest consistency. If no primary is available, raise
300  :class:`~pymongo.errors.AutoReconnect`.
301 
302  - ``PRIMARY_PREFERRED``: Read from the primary if available, or if there is
303  none, read from a secondary.
304 
305  - ``SECONDARY``: Read from a secondary. If no secondary is available,
306  raise :class:`~pymongo.errors.AutoReconnect`.
307 
308  - ``SECONDARY_PREFERRED``: Read from a secondary if available, otherwise
309  from the primary.
310 
311  - ``NEAREST``: Read from any member.
312 
313  :class:`~pymongo.mongo_client.MongoClient` connected to a mongos, with a
314  sharded cluster of replica sets:
315 
316  - ``PRIMARY``: Read from the primary of the shard, or raise
317  :class:`~pymongo.errors.OperationFailure` if there is none.
318  This is the default.
319 
320  - ``PRIMARY_PREFERRED``: Read from the primary of the shard, or if there is
321  none, read from a secondary of the shard.
322 
323  - ``SECONDARY``: Read from a secondary of the shard, or raise
324  :class:`~pymongo.errors.OperationFailure` if there is none.
325 
326  - ``SECONDARY_PREFERRED``: Read from a secondary of the shard if available,
327  otherwise from the shard primary.
328 
329  - ``NEAREST``: Read from any shard member.
330  """
331  PRIMARY = Primary()
332  PRIMARY_PREFERRED = PrimaryPreferred()
333  SECONDARY = Secondary()
334  SECONDARY_PREFERRED = SecondaryPreferred()
335  NEAREST = Nearest()

344 class MovingAverage(object):
345  """Tracks an exponentially-weighted moving average."""
346  def __init__(self):
347  self.average = None
348 
349  def add_sample(self, sample):
350  if sample < 0:
351  # Likely system time change while waiting for ismaster response
352  # and not using time.monotonic. Ignore it, the next one will
353  # probably be valid.
354  return
355  if self.average is None:
356  self.average = sample
357  else:
358  # The Server Selection Spec requires an exponentially weighted
359  # average with alpha = 0.2.
360  self.average = 0.8 * self.average + 0.2 * sample
361 
362  def get(self):
363  """Get the calculated average, or None if no samples yet."""
364  return self.average
365 
366  def reset(self):
367  self.average = None
