61  def select_servers(self,
62  selector,
63  server_selection_timeout=None,
64  address=None):
65  """Return a list of Servers matching selector, or time out.
66 
67  :Parameters:
68  - `selector`: function that takes a list of Servers and returns
69  a subset of them.
70  - `server_selection_timeout` (optional): maximum seconds to wait.
71  If not provided, the default value common.SERVER_SELECTION_TIMEOUT
72  is used.
73  - `address`: optional server address to select.
74 
75  Calls self.open() if needed.
76 
77  Raises exc:`ServerSelectionTimeoutError` after
78  `server_selection_timeout` if no matching servers are found.
79  """
80  if server_selection_timeout is None:
81  server_timeout = self._settings.server_selection_timeout
82  else:
83  server_timeout = server_selection_timeout
84 
85  with self._lock:
86  self._description.check_compatible()
87 
88  now = _time()
89  end_time = now + server_timeout
90  server_descriptions = self._apply_selector(selector, address)
91 
92  while not server_descriptions:
93  # No suitable servers.
94  if server_timeout == 0 or now > end_time:
95  raise ServerSelectionTimeoutError(
96  self._error_message(selector))
97 
98  self._ensure_opened()
99  self._request_check_all()
100 
101  # Release the lock and wait for the topology description to
102  # change, or for a timeout. We won't miss any changes that
103  # came after our most recent _apply_selector call, since we've
104  # held the lock until now.
105  self._condition.wait(common.MIN_HEARTBEAT_INTERVAL)
106  self._description.check_compatible()
107  now = _time()
108  server_descriptions = self._apply_selector(selector, address)
109 
110  return [self.get_server_by_address(sd.address)
111  for sd in server_descriptions]

145  def on_change(self, server_description):
146  """Process a new ServerDescription after an ismaster call completes."""
147  # We do no I/O holding the lock.
148  with self._lock:
149  # Any monitored server was definitely in the topology description
150  # once. Check if it's still in the description or if some state-
151  # change removed it. E.g., we got a host list from the primary
152  # that didn't include this server.
153  if self._description.has_server(server_description.address):
154  self._description = updated_topology_description(
155  self._description, server_description)
156 
157  self._update_servers()
158 
159  # Wake waiters in select_servers().

306  def _apply_selector(self, selector, address):
307  if self._description.topology_type == TOPOLOGY_TYPE.Single:
308  # Ignore the selector.
309  return self._description.known_servers
310  elif address:
311  sd = self._description.server_descriptions().get(address)
312  return [sd] if sd else []
313  elif self._description.topology_type == TOPOLOGY_TYPE.Sharded:
314  return apply_local_threshold(self._settings.local_threshold_ms,
315  self._description.known_servers)
316  else:
317  sds = selector(self._description.known_servers)
318  return apply_local_threshold(
319  self._settings.local_threshold_ms, sds)

321  def _update_servers(self):
322  """Sync our Servers from TopologyDescription.server_descriptions.
323 
324  Hold the lock while calling this.
325  """
326  for address, sd in self._description.server_descriptions().items():
327  if address not in self._servers:
328  monitor = self._settings.monitor_class(
329  server_description=sd,
330  topology=self,
331  pool=self._create_pool_for_monitor(address),
332  topology_settings=self._settings)
333 
334  server = Server(
335  server_description=sd,
336  pool=self._create_pool_for_server(address),
337  monitor=monitor)
338 
339  self._servers[address] = server
340  server.open()
341  else:
342  self._servers[address].description = sd
343 
344  for address, server in list(self._servers.items()):
345  if not self._description.has_server(address):
346  server.close()
347  self._servers.pop(address)

368  def _error_message(self, selector):
369  """Format an error message if server selection fails.
370 
371  Hold the lock when calling this.
372  """
373  is_replica_set = self._description.topology_type in (
374  TOPOLOGY_TYPE.ReplicaSetWithPrimary,
375  TOPOLOGY_TYPE.ReplicaSetNoPrimary)
376 
377  if is_replica_set:
378  server_plural = 'replica set members'
379  elif self._description.topology_type == TOPOLOGY_TYPE.Sharded:
380  server_plural = 'mongoses'
381  else:
382  server_plural = 'servers'
383 
384  if self._description.known_servers:
385  # We've connected, but no servers match the selector.
386  if selector is writable_server_selector:
387  if is_replica_set:
388  return 'No primary available for writes'
389  else:
390  return 'No %s available for writes' % server_plural
391  else:
392  return 'No %s match selector "%s"' % (server_plural, selector)
393  else:
394  addresses = list(self._description.server_descriptions())
395  servers = list(self._description.server_descriptions().values())
396  if not servers:
397  if is_replica_set:
398  # We removed all servers because of the wrong setName?
399  return 'No %s available for replica set name "%s"' % (
400  server_plural, self._settings.replica_set_name)
401  else:
402  return 'No %s available' % server_plural
403 
404  # 1 or more servers, all Unknown. Are they unknown for one reason?
405  error = servers[0].error
406  same = all(server.error == error for server in servers[1:])
407  if same:
408  if error is None:
409  # We're still discovering.
410  return 'No %s found yet' % server_plural
411 
412  if (is_replica_set and not
413  set(addresses).intersection(self._seed_addresses)):
414  # We replaced our seeds with new hosts but can't reach any.
415  return (
416  'Could not reach any servers in %s. Replica set is'
417  ' configured with internal hostnames or IPs?' %
418  addresses)
419 
420  return str(error)
421  else:
422  return ','.join(str(server.error) for server in servers
423  if server.error)
