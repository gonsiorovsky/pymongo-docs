26 def command(sock, dbname, spec, slave_ok, is_mongos, read_preference,
27  codec_options, check=True, allowable_errors=None):
28  """Execute a command over the socket, or raise socket.error.
29 
30  :Parameters:
31  - `sock`: a raw socket instance
32  - `dbname`: name of the database on which to run the command
33  - `spec`: a command document as a dict, SON, or mapping object
34  - `slave_ok`: whether to set the SlaveOkay wire protocol bit
35  - `is_mongos`: are we connected to a mongos?
36  - `read_preference`: a read preference
37  - `codec_options`: a CodecOptions instance
38  - `check`: raise OperationFailure if there are errors
39  - `allowable_errors`: errors to ignore if `check` is True
40  """
41  ns = dbname + '.$cmd'
42  flags = 4 if slave_ok else 0
43  if is_mongos:
44  spec = message._maybe_add_read_preference(spec, read_preference)
45  request_id, msg, _ = message.query(flags, ns, 0, -1, spec,
46  None, codec_options)
47  sock.sendall(msg)
48  response = receive_message(sock, 1, request_id)
49  unpacked = helpers._unpack_response(response, codec_options=codec_options)
50  response_doc = unpacked['data'][0]
51  msg = "command %s on namespace %s failed: %%s" % (
52  repr(spec).replace("%", "%%"), ns)
53  if check:
54  helpers._check_command_response(response_doc, msg, allowable_errors)
55  return response_doc

58 def receive_message(sock, operation, request_id):
59  """Receive a raw BSON message or raise socket.error."""
60  header = _receive_data_on_socket(sock, 16)
61  length = _UNPACK_INT(header[:4])[0]
62 
63  # No request_id for exhaust cursor "getMore".
64  if request_id is not None:
65  response_id = _UNPACK_INT(header[8:12])[0]
66  assert request_id == response_id, "ids don't match %r %r" % (
67  request_id, response_id)
68 
69  assert operation == _UNPACK_INT(header[12:])[0]
70  return _receive_data_on_socket(sock, length - 16)

73 def _receive_data_on_socket(sock, length):
74  msg = b""
75  while length:
76  chunk = sock.recv(length)
77  if chunk == b"":
78  raise AutoReconnect("connection closed")
79 
80  length -= len(chunk)
81  msg += chunk
82 
83  return msg
