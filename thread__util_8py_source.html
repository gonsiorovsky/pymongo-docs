27 ### Begin backport from CPython 3.2 for timeout support for Semaphore.acquire
28 class Semaphore:
29 
30  # After Tim Peters' semaphore class, but not quite the same (no maximum)
31 
32  def __init__(self, value=1):
33  if value < 0:
34  raise ValueError("semaphore initial value must be >= 0")
35  self._cond = threading.Condition(threading.Lock())
36  self._value = value
37 
38  def acquire(self, blocking=True, timeout=None):
39  if not blocking and timeout is not None:
40  raise ValueError("can't specify timeout for non-blocking acquire")
41  rc = False
42  endtime = None
43  self._cond.acquire()
44  while self._value == 0:
45  if not blocking:
46  break
47  if timeout is not None:
48  if endtime is None:
49  endtime = _time() + timeout
50  else:
51  timeout = endtime - _time()
52  if timeout <= 0:
53  break
54  self._cond.wait(timeout)
55  else:
56  self._value = self._value - 1
57  rc = True
58  self._cond.release()
59  return rc
60 
61  __enter__ = acquire
62 
63  def release(self):
64  self._cond.acquire()
65  self._value = self._value + 1
66  self._cond.notify()
67  self._cond.release()
68 
69  def __exit__(self, t, v, tb):
70  self.release()

72  @property
73  def counter(self):
74  return self._value

134 class Event(object):
135  """Copy of standard threading.Event, but uses a custom condition class.
136 
137  Allows async frameworks to override monitors' synchronization behavior
138  with TopologySettings.condition_class.
139 
140  Copied from CPython's threading.py at hash c7960cc9.
141  """
142 
143  def __init__(self, condition_class):
144  self._cond = condition_class(threading.Lock())
145  self._flag = False
146 
147  def is_set(self):
148  """Return true if and only if the internal flag is true."""
149  return self._flag
150

153  def set(self):
154  """Set the internal flag to true.
155 
156  All threads waiting for it to become true are awakened. Threads
157  that call wait() once the flag is true will not block at all.
158 
159  """
160  self._cond.acquire()
161  try:
162  self._flag = True
163  self._cond.notify_all()

165  self._cond.release()

168  """Reset the internal flag to false.
169 
170  Subsequently, threads calling wait() will block until set() is called to
171  set the internal flag to true again.
172 
173  """
174  self._cond.acquire()
175  try:
176  self._flag = False
177  finally:
178  self._cond.release()
179 
180  def wait(self, timeout=None):
181  """Block until the internal flag is true.
182 
183  If the internal flag is true on entry, return immediately. Otherwise,
184  block until another thread calls set() to set the flag to true, or until
185  the optional timeout occurs.
186 
187  When the timeout argument is present and not None, it should be a
188  floating point number specifying a timeout for the operation in seconds
189  (or fractions thereof).
190 
191  This method returns the internal flag on exit, so it will always return
192  True except if a timeout is given and the operation times out.
193 
194  """
195  self._cond.acquire()
196  try:
197  signaled = self._flag
198  if not signaled:
199  signaled = self._cond.wait(timeout)
200  return signaled
201  finally:
202  self._cond.release()
