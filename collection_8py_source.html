319  def bulk_write(self, requests, ordered=True):
320  """Send a batch of write operations to the server.
321 
322  Requests are passed as a list of write operation instances (
323  :class:`~pymongo.operations.InsertOne`,
324  :class:`~pymongo.operations.UpdateOne`,
325  :class:`~pymongo.operations.UpdateMany`,
326  :class:`~pymongo.operations.ReplaceOne`,
327  :class:`~pymongo.operations.DeleteOne`, or
328  :class:`~pymongo.operations.DeleteMany`).
329

331  ... print(doc)
332  ...
333  {u'x': 1, u'_id': ObjectId('54f62e60fba5226811f634ef')}

336  ...

338  >>> requests = [InsertOne({'y': 1}), DeleteOne({'x': 1}),

340  >>> result = db.test.bulk_write(requests)
341  >>> result.inserted_count
342  1
343  >>> result.deleted_count
344  1
345  >>> result.modified_count
346  0
347  >>> result.upserted_ids
348  {2: ObjectId('54f62ee28891e756a6e1abd5')}

351  ...
352  {u'x': 1, u'_id': ObjectId('54f62e60fba5226811f634f0')}
353  {u'y': 1, u'_id': ObjectId('54f62ee2fba5226811f634f1')}
354  {u'z': 1, u'_id': ObjectId('54f62ee28891e756a6e1abd5')}

357  - `requests`: A list of write operations (see examples above).
358  - `ordered` (optional): If ``True`` (the default) requests will be

383  def _insert(self, sock_info, docs, ordered=True,
384  check_keys=True, manipulate=False, write_concern=None):
385  """Internal insert helper."""
386  return_one = False
387  if isinstance(docs, collections.MutableMapping):
388  return_one = True
389  docs = [docs]
390 
391  ids = []
392 
393  if manipulate:
394  def gen():
395  """Generator that applies SON manipulators to each document
396  and adds _id if necessary.
397  """
398  _db = self.__database
399  for doc in docs:
400  # Apply user-configured SON manipulators. This order of
401  # operations is required for backwards compatibility,
402  # see PYTHON-709.
403  doc = _db._apply_incoming_manipulators(doc, self)
404  if '_id' not in doc:
405  doc['_id'] = ObjectId()
406 
407  doc = _db._apply_incoming_copying_manipulators(doc, self)
408  ids.append(doc['_id'])
409  yield doc
410  else:
411  def gen():
412  """Generator that only tracks existing _ids."""
413  for doc in docs:
414  ids.append(doc.get('_id'))
415  yield doc
416 
417  concern = (write_concern or self.write_concern).document
418  safe = concern.get("w") != 0
419 
420  if sock_info.max_wire_version > 1 and safe:
421  # Insert command.
422  command = SON([('insert', self.name),
423  ('ordered', ordered)])
424 
425  if concern:
426  command['writeConcern'] = concern
427 
428  results = message._do_batched_write_command(
429  self.database.name + ".$cmd", _INSERT, command,
430  gen(), check_keys, self.codec_options, sock_info)
431  _check_write_command_response(results)
432  else:
433  # Legacy batched OP_INSERT.
434  message._do_batched_insert(self.__full_name, gen(), check_keys,
435  safe, concern, not ordered,
436  self.codec_options, sock_info)
437  if return_one:
438  return ids[0]
439  else:
440  return ids

511  def _update(self, sock_info, filter, document, upsert=False,
512  check_keys=True, multi=False, manipulate=False,
513  write_concern=None):
514  """Internal update / replace helper."""
515  common.validate_is_mapping("filter", filter)
516  common.validate_boolean("upsert", upsert)
517  if manipulate:
518  document = self.__database._fix_incoming(document, self)
519 
520  concern = (write_concern or self.write_concern).document
521  safe = concern.get("w") != 0
522 
523  if sock_info.max_wire_version > 1 and safe:
524  # Update command.
525  command = SON([('update', self.name)])
526  if concern:
527  command['writeConcern'] = concern
528 
529  docs = [SON([('q', filter), ('u', document),
530  ('multi', multi), ('upsert', upsert)])]
531 
532  results = message._do_batched_write_command(
533  self.database.name + '.$cmd', _UPDATE, command,
534  docs, check_keys, self.codec_options, sock_info)
535  _check_write_command_response(results)
536 
537  _, result = results[0]
538  # Add the updatedExisting field for compatibility.
539  if result.get('n') and 'upserted' not in result:
540  result['updatedExisting'] = True
541  else:
542  result['updatedExisting'] = False
543  # MongoDB >= 2.6.0 returns the upsert _id in an array
544  # element. Break it out for backward compatibility.
545  if 'upserted' in result:
546  result['upserted'] = result['upserted'][0]['_id']
547 
548  return result
549 
550  else:
551  # Legacy OP_UPDATE.
552  request_id, msg, max_size = message.update(
553  self.__full_name, upsert, multi, filter, document, safe,
554  concern, check_keys, self.codec_options)
555  return sock_info.legacy_write(request_id, msg, max_size, safe)
