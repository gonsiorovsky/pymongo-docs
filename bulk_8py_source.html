98 def _merge_legacy(run, full_result, result, index):
99  """Merge a result from a legacy opcode into the full results.
100  """
101  affected = result.get('n', 0)
102 
103  errmsg = result.get("errmsg", result.get("err", ""))
104  if errmsg:
105  # wtimeout is not considered a hard failure in
106  # MongoDB 2.6 so don't treat it like one here.
107  if result.get("wtimeout"):
108  error_doc = {'errmsg': errmsg, 'code': _WRITE_CONCERN_ERROR}
109  full_result['writeConcernErrors'].append(error_doc)
110  else:
111  code = result.get("code", _UNKNOWN_ERROR)
112  error = _make_error(run.index(index), code, errmsg, run.ops[index])
113  if "errInfo" in result:
114  error["errInfo"] = result["errInfo"]
115  full_result["writeErrors"].append(error)
116  return
117  if run.op_type == _INSERT:
118  full_result['nInserted'] += 1
119  elif run.op_type == _UPDATE:
120  if "upserted" in result:
121  doc = {_UINDEX: run.index(index), _UID: result["upserted"]}
122  full_result["upserted"].append(doc)
123  full_result['nUpserted'] += affected
124  # Versions of MongoDB before 2.6 don't return the _id for an
125  # upsert if _id is not an ObjectId.
126  elif result.get("updatedExisting") == False and affected == 1:
127  op = run.ops[index]
128  # If _id is in both the update document *and* the query spec
129  # the update document _id takes precedence.
130  _id = op['u'].get('_id', op['q'].get('_id'))
131  doc = {_UINDEX: run.index(index), _UID: _id}
132  full_result["upserted"].append(doc)
133  full_result['nUpserted'] += affected
134  else:
135  full_result['nMatched'] += affected
136 
137  elif run.op_type == _DELETE:
138  full_result['nRemoved'] += affected
139

141 def _merge_command(run, full_result, results):
142  """Merge a group of results from write commands into the full result.
143  """
144  for offset, result in results:
145 
146  affected = result.get("n", 0)
147 
148  if run.op_type == _INSERT:
149  full_result["nInserted"] += affected
150 
151  elif run.op_type == _DELETE:
152  full_result["nRemoved"] += affected
153 
154  elif run.op_type == _UPDATE:
155  upserted = result.get("upserted")
156  if upserted:
157  if isinstance(upserted, list):
158  n_upserted = len(upserted)
159  for doc in upserted:
160  doc["index"] = run.index(doc["index"] + offset)
161  full_result["upserted"].extend(upserted)
162  else:
163  n_upserted = 1
164  index = run.index(offset)
165  doc = {_UINDEX: index, _UID: upserted}
166  full_result["upserted"].append(doc)
167  full_result["nUpserted"] += n_upserted
168  full_result["nMatched"] += (affected - n_upserted)
169  else:
170  full_result["nMatched"] += affected
171  n_modified = result.get("nModified")
172  # SERVER-13001 - in a mixed sharded cluster a call to
173  # update could return nModified (>= 2.6) or not (<= 2.4).
174  # If any call does not return nModified we can't report
175  # a valid final count so omit the field completely.
176  if n_modified is not None and "nModified" in full_result:
177  full_result["nModified"] += n_modified
178  else:
179  full_result.pop("nModified", None)
180 
181  write_errors = result.get("writeErrors")
182  if write_errors:
183  for doc in write_errors:
184  idx = doc["index"] + offset
185  doc["index"] = run.index(idx)
186  # Add the failed operation to the error document.
187  doc[_UOP] = run.ops[idx]
188  full_result["writeErrors"].extend(write_errors)
189 
190  wc_error = result.get("writeConcernError")
191  if wc_error:
192  full_result["writeConcernErrors"].append(wc_error)

195 class _Bulk(object):
196  """The private guts of the bulk write API.
197  """
198  def __init__(self, collection, ordered):
199  """Initialize a _Bulk instance.
200  """
201  self.collection = collection
202  self.ordered = ordered

205  self.namespace = collection.database.name + '.$cmd'

208  def add_insert(self, document):
209  """Add an insert document to the list of ops.
210  """
211  validate_is_mutable_mapping("document", document)
212  # Generate ObjectId client side.
213  if '_id' not in document:
214  document['_id'] = ObjectId()
215  self.ops.append((_INSERT, document))
216 
217  def add_update(self, selector, update, multi=False, upsert=False):
218  """Create an update document and add it to the list of ops.
219  """
220  validate_ok_for_update(update)
221  cmd = SON([('q', selector), ('u', update),
222  ('multi', multi), ('upsert', upsert)])
223  self.ops.append((_UPDATE, cmd))
224 
225  def add_replace(self, selector, replacement, upsert=False):
226  """Create a replace document and add it to the list of ops.
227  """
228  validate_ok_for_replace(replacement)
229  cmd = SON([('q', selector), ('u', replacement),
230  ('multi', False), ('upsert', upsert)])
231  self.ops.append((_UPDATE, cmd))
232 
233  def add_delete(self, selector, limit):
234  """Create a delete document and add it to the list of ops.
235  """
236  cmd = SON([('q', selector), ('limit', limit)])
237  self.ops.append((_DELETE, cmd))

239  def gen_ordered(self):
240  """Generate batches of operations, batched by type of
241  operation, in the order **provided**.
242  """
243  run = None
244  for idx, (op_type, operation) in enumerate(self.ops):
245  if run is None:
246  run = _Run(op_type)
247  elif run.op_type != op_type:
248  yield run
249  run = _Run(op_type)
250  run.add(idx, operation)
251  yield run
252 
253  def gen_unordered(self):
254  """Generate batches of operations, batched by type of
255  operation, in arbitrary order.
256  """
257  operations = [_Run(_INSERT), _Run(_UPDATE), _Run(_DELETE)]
258  for idx, (op_type, operation) in enumerate(self.ops):
259  operations[op_type].add(idx, operation)
260 
261  for run in operations:
262  if run.ops:
263  yield run

265  def execute_command(self, sock_info, generator, write_concern):
266  """Execute using write commands.
267  """
268  # nModified is only reported for write commands, not legacy ops.
269  full_result = {
270  "writeErrors": [],
271  "writeConcernErrors": [],
272  "nInserted": 0,
273  "nUpserted": 0,
274  "nMatched": 0,
275  "nModified": 0,
276  "nRemoved": 0,
277  "upserted": [],
278  }
279  for run in generator:
280  cmd = SON([(_COMMANDS[run.op_type], self.collection.name),
281  ('ordered', self.ordered)])
282  if write_concern.document:
283  cmd['writeConcern'] = write_concern.document
284 
285  results = _do_batched_write_command(
286  self.namespace, run.op_type, cmd,
287  run.ops, True, self.collection.codec_options, sock_info)
288 
289  _merge_command(run, full_result, results)
290  # We're supposed to continue if errors are
291  # at the write concern level (e.g. wtimeout)
292  if self.ordered and full_result['writeErrors']:
293  break
294 
295  if full_result["writeErrors"] or full_result["writeConcernErrors"]:
296  if full_result['writeErrors']:
297  full_result['writeErrors'].sort(
298  key=lambda error: error['index'])
299  raise BulkWriteError(full_result)
300  return full_result
301 
302  def execute_no_results(self, sock_info, generator):
303  """Execute all operations, returning no results (w=0).
304  """
305  coll = self.collection
306  # If ordered is True we have to send GLE or use write
307  # commands so we can abort on the first error.
308  write_concern = WriteConcern(w=int(self.ordered))
309 
310  for run in generator:
311  try:
312  if run.op_type == _INSERT:
313  coll._insert(sock_info,
314  run.ops,
315  self.ordered,
316  write_concern=write_concern)
317  elif run.op_type == _UPDATE:
318  for operation in run.ops:
319  doc = operation['u']
320  check_keys = True
321  if doc and next(iter(doc)).startswith('$'):
322  check_keys = False
323  coll._update(sock_info,
324  operation['q'],
325  doc,
326  operation['upsert'],
327  check_keys,
328  operation['multi'],
329  write_concern=write_concern)
330  else:
331  for operation in run.ops:
332  coll._delete(sock_info,
333  operation['q'],
334  not operation['limit'],
335  write_concern)
336  except OperationFailure:
337  if self.ordered:
338  break
339 
340  def execute_legacy(self, sock_info, generator, write_concern):
341  """Execute using legacy wire protocol ops.
342  """
343  coll = self.collection
344  full_result = {
345  "writeErrors": [],
346  "writeConcernErrors": [],
347  "nInserted": 0,
348  "nUpserted": 0,
349  "nMatched": 0,
350  "nRemoved": 0,
351  "upserted": [],
352  }
353  stop = False
354  for run in generator:
355  for idx, operation in enumerate(run.ops):
356  try:
357  # To do per-operation reporting we have to do ops one
358  # at a time. That means the performance of bulk insert
359  # will be slower here than calling Collection.insert()
360  if run.op_type == _INSERT:
361  coll._insert(sock_info,
362  operation,
363  write_concern=write_concern)
364  result = {}
365  elif run.op_type == _UPDATE:
366  doc = operation['u']
367  check_keys = True
368  if doc and next(iter(doc)).startswith('$'):
369  check_keys = False
370  result = coll._update(sock_info,
371  operation['q'],
372  doc,
373  operation['upsert'],
374  check_keys,
375  operation['multi'],
376  write_concern=write_concern)
377  else:
378  result = coll._delete(sock_info,
379  operation['q'],
380  not operation['limit'],
381  write_concern)
382  _merge_legacy(run, full_result, result, idx)
383  except DocumentTooLarge as exc:
384  # MongoDB 2.6 uses error code 2 for "too large".
385  error = _make_error(
386  run.index(idx), _BAD_VALUE, str(exc), operation)
387  full_result['writeErrors'].append(error)
388  if self.ordered:
389  stop = True
390  break
391  except OperationFailure as exc:
392  if not exc.details:
393  # Some error not related to the write operation
394  # (e.g. kerberos failure). Re-raise immediately.
395  raise
396  _merge_legacy(run, full_result, exc.details, idx)
397  # We're supposed to continue if errors are
398  # at the write concern level (e.g. wtimeout)
399  if self.ordered and full_result["writeErrors"]:
400  stop = True
401  break
402  if stop:
403  break
404 
405  if full_result["writeErrors"] or full_result['writeConcernErrors']:
406  if full_result['writeErrors']:
407  full_result['writeErrors'].sort(
408  key=lambda error: error['index'])
409  raise BulkWriteError(full_result)
410  return full_result

412  def execute(self, write_concern):
413  """Execute operations.
414  """
415  if not self.ops:
416  raise InvalidOperation('No operations to execute')
417  if self.executed:
418  raise InvalidOperation('Bulk operations can '
419  'only be executed once.')
420  self.executed = True
421  write_concern = (WriteConcern(**write_concern) if
422  write_concern else self.collection.write_concern)
423 
424  if self.ordered:
425  generator = self.gen_ordered()
426  else:
427  generator = self.gen_unordered()
428 
429  client = self.collection.database.client
430  with client._socket_for_writes() as sock_info:
431  if not write_concern.acknowledged:
432  self.execute_no_results(sock_info, generator)
433  elif sock_info.max_wire_version > 1:
434  return self.execute_command(sock_info, generator, write_concern)
435  else:
436  return self.execute_legacy(sock_info, generator, write_concern)
