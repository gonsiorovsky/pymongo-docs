64  def batch_size(self, batch_size):
65  """Limits the number of documents returned in one batch. Each batch
66  requires a round trip to the server. It can be adjusted to optimize
67  performance and limit data transfer.
68 
69  .. note:: batch_size can not override MongoDB's internal limits on the
70  amount of data it will return to the client in a single batch (i.e
71  if you set batch size to 1,000,000,000, MongoDB will currently only
72  return 4-16MB of results per batch).
73 
74  Raises :exc:`TypeError` if `batch_size` is not an integer.
75  Raises :exc:`ValueError` if `batch_size` is less than ``0``.
76 
77  :Parameters:
78  - `batch_size`: The size of each batch of results requested.
79  """
80  if not isinstance(batch_size, integer_types):
81  raise TypeError("batch_size must be an integer")
82  if batch_size < 0:
83  raise ValueError("batch_size must be >= 0")
84 
85  self.__batch_size = batch_size == 1 and 2 or batch_size
86  return self

88  def __send_message(self, operation):
89  """Send a getmore message and handle the response.
90  """
91  client = self.__collection.database.client
92  try:
93  response = client._send_message_with_response(
94  operation, address=self.__address)
95  except AutoReconnect:
96  # Don't try to send kill cursors on another socket
97  # or to another server. It can cause a _pinValue
98  # assertion on some server releases if we get here
99  # due to a socket timeout.
100  self.__killed = True
101  raise
102 
103  try:
104  doc = helpers._unpack_response(response.data,
105  self.__id,
106  self.__collection.codec_options)
107  except CursorNotFound:
108  self.__killed = True
109  raise
110  except NotMasterError:
111  # Don't send kill cursors to another server after a "not master"
112  # error. It's completely pointless.
113  self.__killed = True
114  client._reset_server_and_request_check(self.address)
115  raise
116  self.__id = doc["cursor_id"]
117  if self.__id == 0:
118  self.__killed = True
119 
120  assert doc["starting_from"] == self.__retrieved, (
121  "Result batch started from %s, expected %s" % (
122  doc['starting_from'], self.__retrieved))
123 
124  self.__retrieved += doc["number_returned"]
125  self.__data = deque(doc["data"])
