30 class TopologyDescription(object):
31  def __init__(self, topology_type, server_descriptions, replica_set_name):
32  """Represent a topology of servers.
33 
34  :Parameters:
35  - `topology_type`: initial type
36  - `server_descriptions`: dict of (address, ServerDescription) for
37  all seeds
38  - `replica_set_name`: replica set name or None
39  """
40  self._topology_type = topology_type
41  self._replica_set_name = replica_set_name
42  self._server_descriptions = server_descriptions
43 
44  # Is PyMongo compatible with all servers' wire protocols?
45  self._incompatible_err = None
46 
47  for s in self._server_descriptions.values():
48  # s.min/max_wire_version is the server's wire protocol.
49  # MIN/MAX_SUPPORTED_WIRE_VERSION is what PyMongo supports.
50  server_too_new = (
51  # Server too new.
52  s.min_wire_version is not None
53  and s.min_wire_version > common.MAX_SUPPORTED_WIRE_VERSION)
54 
55  server_too_old = (
56  # Server too old.
57  s.max_wire_version is not None
58  and s.max_wire_version < common.MIN_SUPPORTED_WIRE_VERSION)
59 
60  if server_too_new or server_too_old:
61  self._incompatible_err = (
62  "Server at %s:%d "
63  "uses wire protocol versions %d through %d, "
64  "but PyMongo only supports %d through %d"
65  % (s.address[0], s.address[1],
66  s.min_wire_version, s.max_wire_version,
67  common.MIN_SUPPORTED_WIRE_VERSION,
68  common.MAX_SUPPORTED_WIRE_VERSION))
69 
70  break

132 def updated_topology_description(topology_description, server_description):
133  """Return an updated copy of a TopologyDescription.
134 
135  :Parameters:
136  - `topology_description`: the current TopologyDescription
137  - `server_description`: a new ServerDescription that resulted from
138  an ismaster call
139 
140  Called after attempting (successfully or not) to call ismaster on the
141  server at server_description.address. Does not modify topology_description.
142  """
143  address = server_description.address
144 
145  # These values will be updated, if necessary, to form the new
146  # TopologyDescription.
147  topology_type = topology_description.topology_type
148  set_name = topology_description.replica_set_name
149  server_type = server_description.server_type
150 
151  # Don't mutate the original dict of server descriptions; copy it.
152  sds = topology_description.server_descriptions()
153 
154  # Replace this server's description with the new one.
155  sds[address] = server_description
156 
157  if topology_type == TOPOLOGY_TYPE.Single:
158  # Single type never changes.
159  return TopologyDescription(TOPOLOGY_TYPE.Single, sds, set_name)
160 
161  if topology_type == TOPOLOGY_TYPE.Unknown:
162  if server_type == SERVER_TYPE.Standalone:
163  sds.pop(address)
164 
165  elif server_type not in (SERVER_TYPE.Unknown, SERVER_TYPE.RSGhost):
166  topology_type = _SERVER_TYPE_TO_TOPOLOGY_TYPE[server_type]
167 
168  if topology_type == TOPOLOGY_TYPE.Sharded:
169  if server_type not in (SERVER_TYPE.Mongos, SERVER_TYPE.Unknown):
170  sds.pop(address)
171 
172  elif topology_type == TOPOLOGY_TYPE.ReplicaSetNoPrimary:
173  if server_type in (SERVER_TYPE.Standalone, SERVER_TYPE.Mongos):
174  sds.pop(address)
175 
176  elif server_type == SERVER_TYPE.RSPrimary:
177  topology_type, set_name = _update_rs_from_primary(
178  sds, set_name, server_description)
179 
180  elif server_type in (
181  SERVER_TYPE.RSSecondary,
182  SERVER_TYPE.RSArbiter,
183  SERVER_TYPE.RSOther):
184  topology_type, set_name = _update_rs_no_primary_from_member(
185  sds, set_name, server_description)
186 
187  elif topology_type == TOPOLOGY_TYPE.ReplicaSetWithPrimary:
188  if server_type in (SERVER_TYPE.Standalone, SERVER_TYPE.Mongos):
189  sds.pop(address)
190  topology_type = _check_has_primary(sds)
191 
192  elif server_type == SERVER_TYPE.RSPrimary:
193  topology_type, set_name = _update_rs_from_primary(
194  sds, set_name, server_description)
195 
196  elif server_type in (
197  SERVER_TYPE.RSSecondary,
198  SERVER_TYPE.RSArbiter,
199  SERVER_TYPE.RSOther):
200  topology_type = _update_rs_with_primary_from_member(
201  sds, set_name, server_description)
202 
203  else:
204  # Server type is Unknown or RSGhost: did we just lose the primary?
205  topology_type = _check_has_primary(sds)
206 
207  # Return updated copy.
208  return TopologyDescription(topology_type, sds, set_name)

211 def _update_rs_from_primary(sds, replica_set_name, server_description):
212  """Update topology description from a primary's ismaster response.
213 
214  Pass in a dict of ServerDescriptions, current replica set name, and the
215  ServerDescription we are processing.
216 
217  Returns (new topology type, new replica_set_name).
218  """
219  if replica_set_name is None:
220  replica_set_name = server_description.replica_set_name
221 
222  elif replica_set_name != server_description.replica_set_name:
223  # We found a primary but it doesn't have the replica_set_name
224  # provided by the user.
225  sds.pop(server_description.address)
226  return _check_has_primary(sds), replica_set_name
227 
228  # We've heard from the primary. Is it the same primary as before?
229  for server in sds.values():
230  if (server.server_type is SERVER_TYPE.RSPrimary
231  and server.address != server_description.address):
232 
233  # Reset old primary's type to Unknown.
234  sds[server.address] = ServerDescription(server.address)
235 
236  # There can be only one prior primary.
237  break
238 
239  # Discover new hosts from this primary's response.
240  for new_address in server_description.all_hosts:
241  if new_address not in sds:
242  sds[new_address] = ServerDescription(new_address)
243 
244  # Remove hosts not in the response.
245  for addr in set(sds) - server_description.all_hosts:
246  sds.pop(addr)
247 
248  # If the host list differs from the seed list, we may not have a primary
249  # after all.
250  return _check_has_primary(sds), replica_set_name

264  assert replica_set_name is not None
265

267  sds.pop(server_description.address)

287

289  sds.pop(server_description.address)
290  return topology_type, replica_set_name
291

293  # it doesn't report. Only add new servers.

295  if address not in sds:
