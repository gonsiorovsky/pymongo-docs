34 def _partition(entity, sep):
35  """Python2.4 doesn't have a partition method so we provide
36  our own that mimics str.partition from later releases.
37 
38  Split the string at the first occurrence of sep, and return a
39  3-tuple containing the part before the separator, the separator
40  itself, and the part after the separator. If the separator is not
41  found, return a 3-tuple containing the string itself, followed
42  by two empty strings.
43  """
44  parts = entity.split(sep, 1)
45  if len(parts) == 2:
46  return parts[0], sep, parts[1]
47  else:
48  return entity, '', ''

51 def _rpartition(entity, sep):
52  """Python2.4 doesn't have an rpartition method so we provide
53  our own that mimics str.rpartition from later releases.
54 
55  Split the string at the last occurrence of sep, and return a
56  3-tuple containing the part before the separator, the separator
57  itself, and the part after the separator. If the separator is not
58  found, return a 3-tuple containing two empty strings, followed
59  by the string itself.
60  """
61  idx = entity.rfind(sep)
62  if idx == -1:
63  return '', '', entity
64  return entity[:idx], sep, entity[idx + 1:]

67 def parse_userinfo(userinfo):
68  """Validates the format of user information in a MongoDB URI.
69  Reserved characters like ':', '/', '+' and '@' must be escaped
70  following RFC 2396.
71 
72  Returns a 2-tuple containing the unescaped username followed
73  by the unescaped password.
74 
75  :Paramaters:
76  - `userinfo`: A string of the form <username>:<password>
77 
78  .. versionchanged:: 2.2
79  Now uses `urllib.unquote_plus` so `+` characters must be escaped.
80  """
81  if '@' in userinfo or userinfo.count(':') > 1:
82  raise InvalidURI("':' or '@' characters in a username or password "
83  "must be escaped according to RFC 2396.")
84  user, _, passwd = _partition(userinfo, ":")
85  # No password is expected with GSSAPI authentication.
86  if not user:
87  raise InvalidURI("The empty string is not valid username.")
88  user = unquote_plus(user)
89  passwd = unquote_plus(passwd)
90 
91  return user, passwd

94 def parse_ipv6_literal_host(entity, default_port):
95  """Validates an IPv6 literal host:port string.
96 
97  Returns a 2-tuple of IPv6 literal followed by port where
98  port is default_port if it wasn't specified in entity.
99 
100  :Parameters:
101  - `entity`: A string that represents an IPv6 literal enclosed
102  in braces (e.g. '[::1]' or '[::1]:27017').
103  - `default_port`: The port number to use when one wasn't
104  specified in entity.
105  """
106  if entity.find(']') == -1:
107  raise ValueError("an IPv6 address literal must be "
108  "enclosed in '[' and ']' according "
109  "to RFC 2732.")
110  i = entity.find(']:')
111  if i == -1:
112  return entity[1:-1], default_port
113  return entity[1: i], entity[i + 2:]

116 def parse_host(entity, default_port=DEFAULT_PORT):
117  """Validates a host string
118 
119  Returns a 2-tuple of host followed by port where port is default_port
120  if it wasn't specified in the string.
121 
122  :Parameters:
123  - `entity`: A host or host:port string where host could be a
124  hostname or IP address.
125  - `default_port`: The port number to use when one wasn't
126  specified in entity.
127  """
128  host = entity
129  port = default_port
130  if entity[0] == '[':
131  host, port = parse_ipv6_literal_host(entity, default_port)
132  elif entity.find(':') != -1:
133  if entity.count(':') > 1:
134  raise ValueError("Reserved characters such as ':' must be "
135  "escaped according RFC 2396. An IPv6 "
136  "address literal must be enclosed in '[' "
137  "and ']' according to RFC 2732.")
138  host, port = host.split(':', 1)
139  if isinstance(port, string_type):
140  if not port.isdigit():
141  raise ValueError("Port number must be an integer.")
142  port = int(port)
143 
144  # Normalize hostname to lowercase, since DNS is case-insensitive:
145  # http://tools.ietf.org/html/rfc4343
146  # This prevents useless rediscovery if "foo.com" is in the seed list but
147  # "FOO.com" is in the ismaster response.
148  return host.lower(), port

162 def _parse_options(opts, delim):
163  """Helper method for split_options which creates the options dict.
164  Also handles the creation of a list for the URI tag_sets/
165  readpreferencetags portion."""
166  options = {}
167  for opt in opts.split(delim):
168  key, val = opt.split("=")
169  if key.lower() == 'readpreferencetags':
170  options.setdefault('readpreferencetags', []).append(val)
171  else:
172  # str(option) to ensure that a unicode URI results in plain 'str'
173  # option names. 'normalized' is then suitable to be passed as
174  # kwargs in all Python versions.
175  options[str(key)] = val
176  return options

179 def split_options(opts, validate=True):
180  """Takes the options portion of a MongoDB URI, validates each option
181  and returns the options in a dictionary.
182 
183  :Parameters:
184  - `opt`: A string representing MongoDB URI options.
185  - `validate`: If ``True`` (the default), validate and normalize all
186  options.
187  """
188  and_idx = opts.find("&")
189  semi_idx = opts.find(";")
190  try:
191  if and_idx >= 0 and semi_idx >= 0:
192  raise InvalidURI("Can not mix '&' and ';' for option separators.")
193  elif and_idx >= 0:
194  options = _parse_options(opts, "&")
195  elif semi_idx >= 0:
196  options = _parse_options(opts, ";")
197  elif opts.find("=") != -1:
198  options = _parse_options(opts, None)
199  else:
200  raise ValueError
201  except ValueError:
202  raise InvalidURI("MongoDB URI options are key=value pairs.")
203 
204  if validate:
205  return validate_options(options)
206  return options

209 def split_hosts(hosts, default_port=DEFAULT_PORT):
210  """Takes a string of the form host1[:port],host2[:port]... and
211  splits it into (host, port) tuples. If [:port] isn't present the
212  default_port is used.
213 
214  Returns a set of 2-tuples containing the host name (or IP) followed by
215  port number.
216 
217  :Parameters:
218  - `hosts`: A string of the form host1[:port],host2[:port],...
219  - `default_port`: The port number to use when one wasn't specified
220  for a host.
221  """
222  nodes = []
223  for entity in hosts.split(','):
224  if not entity:
225  raise ConfigurationError("Empty host "
226  "(or extra comma in host list).")
227  port = default_port
228  # Unix socket entities don't have ports
229  if entity.endswith('.sock'):
230  port = None
231  nodes.append(parse_host(entity, port))
232  return nodes

235 def parse_uri(uri, default_port=DEFAULT_PORT, validate=True):
236  """Parse and validate a MongoDB URI.
237 
238  Returns a dict of the form::
239 
240  {
241  'nodelist': <list of (host, port) tuples>,
242  'username': <username> or None,
243  'password': <password> or None,
244  'database': <database name> or None,
245  'collection': <collection name> or None,
246  'options': <dict of MongoDB URI options>
247  }
248 
249  :Parameters:
250  - `uri`: The MongoDB URI to parse.
251  - `default_port`: The port number to use when one wasn't specified
252  for a host in the URI.
253  - `validate`: If ``True`` (the default), validate and normalize all
254  options.
255  """
256  if not uri.startswith(SCHEME):
257  raise InvalidURI("Invalid URI scheme: URI "
258  "must begin with '%s'" % (SCHEME,))
259 
260  scheme_free = uri[SCHEME_LEN:]
261 
262  if not scheme_free:
263  raise InvalidURI("Must provide at least one hostname or IP.")
264 
265  nodes = None
266  user = None
267  passwd = None
268  dbase = None
269  collection = None
270  options = {}
271 
272  # Check for unix domain sockets in the uri
273  if '.sock' in scheme_free:
274  host_part, _, path_part = _rpartition(scheme_free, '/')
275  try:
276  parse_uri('%s%s' % (SCHEME, host_part))
277  except (ConfigurationError, InvalidURI):
278  host_part = scheme_free
279  path_part = ""
280  else:
281  host_part, _, path_part = _partition(scheme_free, '/')
282 
283  if not path_part and '?' in host_part:
284  raise InvalidURI("A '/' is required between "
285  "the host list and any options.")
286 
287  if '@' in host_part:
288  userinfo, _, hosts = _rpartition(host_part, '@')
289  user, passwd = parse_userinfo(userinfo)
290  else:
291  hosts = host_part
292 
293  nodes = split_hosts(hosts, default_port=default_port)
294 
295  if path_part:
296 
297  if path_part[0] == '?':
298  opts = path_part[1:]
299  else:
300  dbase, _, opts = _partition(path_part, '?')
301  if '.' in dbase:
302  dbase, collection = dbase.split('.', 1)
303 
304  if opts:
305  options = split_options(opts, validate)
306 
307  return {
308  'nodelist': nodes,
309  'username': user,
310  'password': passwd,
311  'database': dbase,
312  'collection': collection,
313  'options': options
314  }
