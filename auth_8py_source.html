140 def _authenticate_scram_sha1(credentials, sock_info):
141  """Authenticate using SCRAM-SHA-1."""
142  username = credentials.username
143  password = credentials.password
144  source = credentials.source
145 
146  # Make local
147  _hmac = hmac.HMAC
148  _sha1 = sha1
149 
150  user = username.encode("utf-8").replace(b"=", b"=3D").replace(b",", b"=2C")
151  nonce = standard_b64encode(
152  (("%s" % (SystemRandom().random(),))[2:]).encode("utf-8"))
153  first_bare = b"n=" + user + b",r=" + nonce
154 
155  cmd = SON([('saslStart', 1),
156  ('mechanism', 'SCRAM-SHA-1'),
157  ('payload', Binary(b"n,," + first_bare)),
158  ('autoAuthorize', 1)])
159  res = sock_info.command(source, cmd)
160 
161  server_first = res['payload']
162  parsed = _parse_scram_response(server_first)
163  iterations = int(parsed[b'i'])
164  salt = parsed[b's']
165  rnonce = parsed[b'r']
166  if not rnonce.startswith(nonce):
167  raise OperationFailure("Server returned an invalid nonce.")
168 
169  without_proof = b"c=biws,r=" + rnonce
170  salted_pass = _hi(_password_digest(username, password).encode("utf-8"),
171  standard_b64decode(salt),
172  iterations)
173  client_key = _hmac(salted_pass, b"Client Key", _sha1).digest()
174  stored_key = _sha1(client_key).digest()
175  auth_msg = b",".join((first_bare, server_first, without_proof))
176  client_sig = _hmac(stored_key, auth_msg, _sha1).digest()
177  client_proof = b"p=" + standard_b64encode(_xor(client_key, client_sig))
178  client_final = b",".join((without_proof, client_proof))
179 
180  server_key = _hmac(salted_pass, b"Server Key", _sha1).digest()
181  server_sig = standard_b64encode(
182  _hmac(server_key, auth_msg, _sha1).digest())
183 
184  cmd = SON([('saslContinue', 1),
185  ('conversationId', res['conversationId']),
186  ('payload', Binary(client_final))])
187  res = sock_info.command(source, cmd)
188 
189  parsed = _parse_scram_response(res['payload'])
190  if parsed[b'v'] != server_sig:
191  raise OperationFailure("Server returned an invalid signature.")
192 
193  # Depending on how it's configured, Cyrus SASL (which the server uses)
194  # requires a third empty challenge.
195  if not res['done']:
196  cmd = SON([('saslContinue', 1),
197  ('conversationId', res['conversationId']),
198  ('payload', Binary(b''))])
199  res = sock_info.command(source, cmd)
200  if not res['done']:
201  raise OperationFailure('SASL conversation failed to complete.')

204 def _password_digest(username, password):
205  """Get a password digest to use for authentication.
206  """
207  if not isinstance(password, string_type):
208  raise TypeError("password must be an "
209  "instance of %s" % (string_type.__name__,))
210  if len(password) == 0:
211  raise ValueError("password can't be empty")
212  if not isinstance(username, string_type):
213  raise TypeError("password must be an "
214  "instance of %s" % (string_type.__name__,))
215 
216  md5hash = md5()
217  data = "%s:mongo:%s" % (username, password)
218  md5hash.update(data.encode('utf-8'))
219  return _unicode(md5hash.hexdigest())

232 def _authenticate_gssapi(credentials, sock_info):
233  """Authenticate using GSSAPI.
234  """
235  if not HAVE_KERBEROS:
236  raise ConfigurationError('The "kerberos" module must be '
237  'installed to use GSSAPI authentication.')
238 
239  try:
240  username = credentials.username
241  gsn = credentials.mechanism_properties.service_name
242  # Starting here and continuing through the while loop below - establish
243  # the security context. See RFC 4752, Section 3.1, first paragraph.
244  result, ctx = kerberos.authGSSClientInit(
245  gsn + '@' + sock_info.host, gssflags=kerberos.GSS_C_MUTUAL_FLAG)
246 
247  if result != kerberos.AUTH_GSS_COMPLETE:
248  raise OperationFailure('Kerberos context failed to initialize.')
249 
250  try:
251  # pykerberos uses a weird mix of exceptions and return values
252  # to indicate errors.
253  # 0 == continue, 1 == complete, -1 == error
254  # Only authGSSClientStep can return 0.
255  if kerberos.authGSSClientStep(ctx, '') != 0:
256  raise OperationFailure('Unknown kerberos '
257  'failure in step function.')
258 
259  # Start a SASL conversation with mongod/s
260  # Note: pykerberos deals with base64 encoded byte strings.
261  # Since mongo accepts base64 strings as the payload we don't
262  # have to use bson.binary.Binary.
263  payload = kerberos.authGSSClientResponse(ctx)
264  cmd = SON([('saslStart', 1),
265  ('mechanism', 'GSSAPI'),
266  ('payload', payload),
267  ('autoAuthorize', 1)])
268  response = sock_info.command('$external', cmd)
269 
270  # Limit how many times we loop to catch protocol / library issues
271  for _ in range(10):
272  result = kerberos.authGSSClientStep(ctx,
273  str(response['payload']))
274  if result == -1:
275  raise OperationFailure('Unknown kerberos '
276  'failure in step function.')
277 
278  payload = kerberos.authGSSClientResponse(ctx) or ''
279 
280  cmd = SON([('saslContinue', 1),
281  ('conversationId', response['conversationId']),
282  ('payload', payload)])
283  response = sock_info.command('$external', cmd)
284 
285  if result == kerberos.AUTH_GSS_COMPLETE:
286  break
287  else:
288  raise OperationFailure('Kerberos '
289  'authentication failed to complete.')
290 
291  # Once the security context is established actually authenticate.
292  # See RFC 4752, Section 3.1, last two paragraphs.
293  if kerberos.authGSSClientUnwrap(ctx,
294  str(response['payload'])) != 1:
295  raise OperationFailure('Unknown kerberos '
296  'failure during GSS_Unwrap step.')
297 
298  if kerberos.authGSSClientWrap(ctx,
299  kerberos.authGSSClientResponse(ctx),
300  username) != 1:
301  raise OperationFailure('Unknown kerberos '
302  'failure during GSS_Wrap step.')
303 
304  payload = kerberos.authGSSClientResponse(ctx)
305  cmd = SON([('saslContinue', 1),
306  ('conversationId', response['conversationId']),
307  ('payload', payload)])
308  sock_info.command('$external', cmd)
309 
310  finally:
311  kerberos.authGSSClientClean(ctx)
312 
313  except kerberos.KrbError as exc:
314  raise OperationFailure(str(exc))

331 def _authenticate_cram_md5(credentials, sock_info):
332  """Authenticate using CRAM-MD5 (RFC 2195)
333  """
334  source = credentials.source
335  username = credentials.username
336  password = credentials.password
337  # The password used as the mac key is the
338  # same as what we use for MONGODB-CR
339  passwd = _password_digest(username, password)
340  cmd = SON([('saslStart', 1),
341  ('mechanism', 'CRAM-MD5'),
342  ('payload', Binary(b'')),
343  ('autoAuthorize', 1)])
344  response = sock_info.command(source, cmd)
345  # MD5 as implicit default digest for digestmod is deprecated
346  # in python 3.4
347  mac = hmac.HMAC(key=passwd.encode('utf-8'), digestmod=md5)
348  mac.update(response['payload'])
349  challenge = username.encode('utf-8') + b' ' + b(mac.hexdigest())
350  cmd = SON([('saslContinue', 1),
351  ('conversationId', response['conversationId']),
352  ('payload', Binary(challenge))])
353  sock_info.command(source, cmd)
