87 def _unpack_response(response, cursor_id=None, codec_options=CodecOptions()):
88  """Unpack a response from the database.
89 
90  Check the response for errors and unpack, returning a dictionary
91  containing the response data.
92 
93  Can raise CursorNotFound, NotMasterError, ExecutionTimeout, or
94  OperationFailure.
95 
96  :Parameters:
97  - `response`: byte string as returned from the database
98  - `cursor_id` (optional): cursor_id we sent to get this response -
99  used for raising an informative exception when we get cursor id not
100  valid at server response
101  - `codec_options` (optional): an instance of
102  :class:`~bson.codec_options.CodecOptions`
103  """
104  response_flag = struct.unpack("<i", response[:4])[0]
105  if response_flag & 1:
106  # Shouldn't get this response if we aren't doing a getMore
107  assert cursor_id is not None
108 
109  raise CursorNotFound("cursor id '%s' not valid at server" %
110  cursor_id)
111  elif response_flag & 2:
112  error_object = bson.BSON(response[20:]).decode()
113  if error_object["$err"].startswith("not master"):
114  raise NotMasterError(error_object["$err"])
115  elif error_object.get("code") == 50:
116  raise ExecutionTimeout(error_object.get("$err"),
117  error_object.get("code"),
118  error_object)
119  raise OperationFailure("database error: %s" %
120  error_object.get("$err"),
121  error_object.get("code"),
122  error_object)
123 
124  result = {}
125  result["cursor_id"] = struct.unpack("<q", response[4:12])[0]
126  result["starting_from"] = struct.unpack("<i", response[12:16])[0]
127  result["number_returned"] = struct.unpack("<i", response[16:20])[0]
128  result["data"] = bson.decode_all(response[20:], codec_options)
129  assert len(result["data"]) == result["number_returned"]
130  return result

133 def _check_command_response(response, msg=None, allowable_errors=None):
134  """Check the response to a command for errors.
135  """
136  if "ok" not in response:
137  # Server didn't recognize our message as a command.
138  raise OperationFailure(response.get("$err"),
139  response.get("code"),
140  response)
141 
142  # TODO: remove, this is moving to _check_gle_response
143  if response.get("wtimeout", False):
144  # MongoDB versions before 1.8.0 return the error message in an "errmsg"
145  # field. If "errmsg" exists "err" will also exist set to None, so we
146  # have to check for "errmsg" first.
147  raise WTimeoutError(response.get("errmsg", response.get("err")),
148  response.get("code"),
149  response)
150 
151  if not response["ok"]:
152 
153  details = response
154  # Mongos returns the error details in a 'raw' object
155  # for some errors.
156  if "raw" in response:
157  for shard in itervalues(response["raw"]):
158  # Grab the first non-empty raw error from a shard.
159  if shard.get("errmsg") and not shard.get("ok"):
160  details = shard
161  break
162 
163  errmsg = details["errmsg"]
164  if allowable_errors is None or errmsg not in allowable_errors:
165 
166  # Server is "not master" or "recovering"
167  if (errmsg.startswith("not master")
168  or errmsg.startswith("node is recovering")):
169  raise NotMasterError(errmsg)
170 
171  # Server assertion failures
172  if errmsg == "db assertion failure":
173  errmsg = ("db assertion failure, assertion: '%s'" %
174  details.get("assertion", ""))
175  raise OperationFailure(errmsg,
176  details.get("assertionCode"),
177  response)
178 
179  # Other errors
180  code = details.get("code")
181  # findAndModify with upsert can raise duplicate key error
182  if code in (11000, 11001, 12582):
183  raise DuplicateKeyError(errmsg, code, response)
184  elif code == 50:
185  raise ExecutionTimeout(errmsg, code, response)
186 
187  msg = msg or "%s"
188  raise OperationFailure(msg % errmsg, code, response)

193  response = _unpack_response(response)

204  # have to check for "errmsg" first.
205  raise WTimeoutError(result.get("errmsg", result.get("err")),

243 def _check_write_command_response(results):
244  """Backward compatibility helper for write command error handling.
245  """
246  errors = [res for res in results
247  if "writeErrors" in res[1] or "writeConcernError" in res[1]]
248  if errors:
249  # If multiple batches had errors
250  # raise from the last batch.
251  offset, result = errors[-1]
252  # Prefer write errors over write concern errors
253  write_errors = result.get("writeErrors")
254  if write_errors:
255  # If the last batch had multiple errors only report
256  # the last error to emulate continue_on_error.
257  error = write_errors[-1]
258  error["index"] += offset
259  if error.get("code") == 11000:
260  raise DuplicateKeyError(error.get("errmsg"), 11000, error)
261  raise WriteError(error.get("errmsg"), error.get("code"), error)
262  else:
