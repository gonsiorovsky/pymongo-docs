368  def command(self, command, value=1, check=True,
369  allowable_errors=None, read_preference=ReadPreference.PRIMARY,
370  codec_options=CodecOptions(), **kwargs):
371  """Issue a MongoDB command.
372 
373  Send command `command` to the database and return the
374  response. If `command` is an instance of :class:`basestring`
375  (:class:`str` in python 3) then the command {`command`: `value`}
376  will be sent. Otherwise, `command` must be an instance of
377  :class:`dict` and will be sent as is.
378 
379  Any additional keyword arguments will be added to the final
380  command document before it is sent.
381 
382  For example, a command like ``{buildinfo: 1}`` can be sent
383  using:
384 
385  >>> db.command("buildinfo")
386 
387  For a command where the value matters, like ``{collstats:
388  collection_name}`` we can do:
389 
390  >>> db.command("collstats", collection_name)
391 
392  For commands that take additional arguments we can use
393  kwargs. So ``{filemd5: object_id, root: file_root}`` becomes:
394 
395  >>> db.command("filemd5", object_id, root=file_root)
396 
397  :Parameters:
398  - `command`: document representing the command to be issued,
399  or the name of the command (for simple commands only).
400 
401  .. note:: the order of keys in the `command` document is
402  significant (the "verb" must come first), so commands
403  which require multiple keys (e.g. `findandmodify`)
404  should use an instance of :class:`~bson.son.SON` or
405  a string and kwargs instead of a Python `dict`.
406 
407  - `value` (optional): value to use for the command verb when
408  `command` is passed as a string
409  - `check` (optional): check the response for errors, raising
410  :class:`~pymongo.errors.OperationFailure` if there are any
411  - `allowable_errors`: if `check` is ``True``, error messages
412  in this list will be ignored by error-checking
413  - `read_preference`: The read preference for this operation.
414  See :mod:`~pymongo.read_preferences` for options.
415  - `codec_options`: A :class:`~bson.codec_options.CodecOptions`
416  instance.
417  - `**kwargs` (optional): additional keyword arguments will
418  be added to the command document before it is sent
419 
420  .. note:: :meth:`command` does **not** obey :attr:`read_preference`
421  or :attr:`codec_options`. You must use the `read_preference` and
422  `codec_options` parameters instead.
423 
424  .. versionchanged:: 3.0
425  Removed the `as_class`, `fields`, `uuid_subtype`, `tag_sets`,
426  and `secondary_acceptable_latency_ms` option.
427  Removed `compile_re` option: PyMongo now always represents BSON
428  regular expressions as :class:`~bson.regex.Regex` objects. Use
429  :meth:`~bson.regex.Regex.try_compile` to attempt to convert from a
430  BSON regular expression to a Python regular expression object.
431  Added the `codec_options` parameter.
432 
433  .. versionchanged:: 2.7
434  Added `compile_re` option. If set to False, PyMongo represented BSON
435  regular expressions as :class:`~bson.regex.Regex` objects instead of
436  attempting to compile BSON regular expressions as Python native
437  regular expressions, thus preventing errors for some incompatible
438  patterns, see `PYTHON-500`_.
439 
440  .. versionchanged:: 2.3
441  Added `tag_sets` and `secondary_acceptable_latency_ms` options.
442  .. versionchanged:: 2.2
443  Added support for `as_class` - the class you want to use for
444  the resulting documents
445 
446  .. _PYTHON-500: https://jira.mongodb.org/browse/PYTHON-500
447 
448  .. mongodoc:: commands
449  """
450  client = self.__client
451  with client._socket_for_reads(read_preference) as (sock_info, slave_ok):
452  return self._command(sock_info, command, slave_ok, value,
453  check, allowable_errors, read_preference,
454  codec_options, **kwargs)

521  def validate_collection(self, name_or_collection,
522  scandata=False, full=False):
523  """Validate a collection.
524 
525  Returns a dict of validation info. Raises CollectionInvalid if
526  validation fails.
527 
528  With MongoDB < 1.9 the result dict will include a `result` key
529  with a string value that represents the validation results. With
530  MongoDB >= 1.9 the `result` key no longer exists and the results
531  are split into individual fields in the result dict.
532 
533  :Parameters:
534  - `name_or_collection`: A Collection object or the name of a
535  collection to validate.
536  - `scandata`: Do extra checks beyond checking the overall
537  structure of the collection.
538  - `full`: Have the server do a more thorough scan of the
539  collection. Use with `scandata` for a thorough scan
540  of the structure of the collection and the individual
541  documents. Ignored in MongoDB versions before 1.9.
542  """
543  name = name_or_collection
544  if isinstance(name, Collection):
545  name = name.name
546 
547  if not isinstance(name, string_type):
548  raise TypeError("name_or_collection must be an instance of "
549  "%s or Collection" % (string_type.__name__,))
550 
551  result = self.command("validate", _unicode(name),
552  scandata=scandata, full=full)
553 
554  valid = True
555  # Pre 1.9 results
556  if "result" in result:
557  info = result["result"]
558  if info.find("exception") != -1 or info.find("corrupt") != -1:
559  raise CollectionInvalid("%s invalid: %s" % (name, info))
560  # Sharded results
561  elif "raw" in result:
562  for _, res in iteritems(result["raw"]):
563  if "result" in res:
564  info = res["result"]
565  if (info.find("exception") != -1 or
566  info.find("corrupt") != -1):
567  raise CollectionInvalid("%s invalid: "
568  "%s" % (name, info))
569  elif not res.get("valid", False):
570  valid = False
571  break
572  # Post 1.9 non-sharded results.
573  elif not result.get("valid", False):
574  valid = False
575 
576  if not valid:
577  raise CollectionInvalid("%s invalid: %r" % (name, result))
578 
579  return result

911  def authenticate(self, name, password=None,
912  source=None, mechanism='DEFAULT', **kwargs):
913  """Authenticate to use this database.
914 
915  Authentication lasts for the life of the underlying client
916  instance, or until :meth:`logout` is called.
917 
918  Raises :class:`TypeError` if (required) `name`, (optional) `password`,

925  in the underlying client instance to be authenticated automatically.
926

928  credentials is not supported. You must call :meth:`logout` before
929  authenticating with new credentials.

932  threads will share the authentication. If you need different
933  authentication profiles for different purposes you must use
934  distinct client instances.
935

938  - `password` (optional): the password of the user to authenticate.

943  :data:`~pymongo.auth.MECHANISMS` for options.
944  By default, use SCRAM-SHA-1 with MongoDB 3.0 and later,
945  MONGODB-CR (MongoDB Challenge Response protocol) for older servers.
946  - `authMechanismProperties` (optional): Used to specify
947  authentication mechanism specific options. To specify the service
948  name for GSSAPI authentication pass
949  authMechanismProperties='SERVICE_NAME:<service name>'

951  .. versionadded:: 2.8
952  Use SCRAM-SHA-1 with MongoDB 3.0 and later.
953 
954  .. versionchanged:: 2.5

956  now raises a subclass of :class:`~pymongo.errors.PyMongoError` if
957  authentication fails due to invalid credentials or configuration
958  issues.
959 
960  .. mongodoc:: authenticate
961  """
962  if not isinstance(name, string_type):
963  raise TypeError("name must be an "
964  "instance of %s" % (string_type.__name__,))
965  if password is not None and not isinstance(password, string_type):
966  raise TypeError("password must be an "
967  "instance of %s" % (string_type.__name__,))

982  password,
983  validated_options)
